压缩文件夹
```bash
for i in */; do zip -r "${i%/}.zip" "$i"; done
```

排除执行文件或文件夹
```bash
zip -r zipfile.zip . -x ".*" -x "app/bower_components" -x "node_modules" -x "*.zip" -x "node_modules"
```
