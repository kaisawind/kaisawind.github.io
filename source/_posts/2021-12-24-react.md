react使用IIFE立即调用函数表达式
<!--more-->

```tsx
  useEffect(() => {
    // Using an IIFE
    (async function asyncGetProduct() {
      const product: Partial<API.Product> = await GetProduct({ id: profile.parent_id });
      if (product.mode) {
        setMode(product.mode)
      }
      if (product.sign) {
        setSign(product.sign)
      }
    })()
  }, [profile.parent_id])
```
