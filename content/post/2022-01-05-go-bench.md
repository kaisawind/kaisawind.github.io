文件`server/load_test.go`
```go
func BenchmarkServer_Load(b *testing.B) {
	b.ResetTimer()
	for i := 0; i < b.N; i++ {
		var reg *[16]byte
		err := pmem.MapAsPOD(ReadAddress, &reg)
		if err != nil {
			b.Error(err)
			b.FailNow()
		}
		// b.Logf("%02X", reg)
	}
}
```

单独测试一条指令
```bash
go test -bench Server_Load
```

在父目录时单独测试一条指令
```bash
go test ./server -bench Server_Load
```
